    let startTime = null;
    let progressTimer = null;

    window.addEventListener('load', () => {
  if (google?.script?.run?.resetProgress)
    google.script.run.resetProgress();
  loadEntities();
  loadDimensionLabels();   // <- NEW

function loadDimensionLabels() {
  if (!google || !google.script || !google.script.run) return;

  google.script.run
    .withSuccessHandler(labels => {
      if (!labels) return;

      // 1) Main tracking category label above the dropdown
      if (labels.dim1Label) {
        const entLabel = document.getElementById('entityLabel');
        if (entLabel) entLabel.textContent = labels.dim1Label;

        // 2) Hint under the Payroll Groups box
        const groupsHint = document.getElementById('groupsHint');
        if (groupsHint) {
          groupsHint.textContent =
            'Choose one or more Payroll Groups under the selected ' + labels.dim1Label + '.';
        }
      }

      // 3) Optional override for the Payroll Groups header itself
      //    If your server DOESN'T return labels.groupLabel, this will just keep the default text.
      if (labels.groupLabel) {
        const gl = document.getElementById('payrollGroupsLabel');
        if (gl) gl.textContent = labels.groupLabel;
      }
    })
    .withFailureHandler(err => {
      console.log('getDimensionLabels error', err);
    })
    .getDimensionLabels();
}



    // Load entities from Masterfile
    function loadEntities() {
      const sel = document.getElementById("entity");
      sel.innerHTML = "<option>Loading…</option>";
      google.script.run
        .withSuccessHandler(entities => {
          sel.innerHTML = "";
          if (!entities || entities.length === 0) {
            sel.innerHTML = "<option value=''>No entities found</option>";
            return;
          }
          entities.forEach(e => {
            const opt = document.createElement("option");
            opt.value = e;
            opt.textContent = e;
            sel.appendChild(opt);
          });
          loadGroupsByEntity();
        })
        .withFailureHandler(err => {
          sel.innerHTML = "<option value=''>Error loading entities</option>";
          alert("Error loading entities:\n" + (err.message || err));
        })
        .getEntities();
    }

    // Load payroll groups based on entity
    function loadGroupsByEntity() {
      const entity = document.getElementById("entity").value || "";
      const groupList = document.getElementById("groupList");
      groupList.innerHTML = "<div>Loading…</div>";

      google.script.run
        .withSuccessHandler(groups => {
          groupList.innerHTML = "";
          if (!groups || groups.length === 0) {
            groupList.innerHTML = "<div>No groups found</div>";
            return;
          }
          groups
            .filter(g => g.toUpperCase() !== "ALL") // remove ALL
            .forEach(g => {
              const row = document.createElement("label");
              row.innerHTML = `<input type="checkbox" class="groupChk" value="${g}"> ${g}`;
              groupList.appendChild(row);
            });
        })
        .withFailureHandler(err => {
          groupList.innerHTML = "<div>Error loading groups</div>";
          alert("Error loading payroll groups:\n" + (err.message || err));
        })
        .getPayrollGroupsByEntity(entity);
    }

    function toggleAllGroups(checked) {
      document.querySelectorAll(".groupChk").forEach(cb => cb.checked = checked);
    }

    function clearAllGroups() {
      document.querySelectorAll(".groupChk").forEach(cb => cb.checked = false);
    }

    function getSelectedGroups() {
      return Array.from(document.querySelectorAll(".groupChk:checked")).map(cb => cb.value);
    }

    function toggleFrequencyUI() {
      const freq = document.getElementById("payrollFrequency").value;
      const codeContainer = document.getElementById("payrollCodeContainer");
      const specialOptions = document.getElementById("specialOptions");
      const selectCode = document.getElementById("payrollCodeSelect");
      const textCode = document.getElementById("payrollCodeText");
      const label = document.getElementById("payrollCodeLabel");

      if (freq === "Semi-Monthly") {
        label.textContent = "Payroll Code (A / B)";
        selectCode.style.display = "block";
        textCode.style.display = "none";
        codeContainer.style.display = "block";
        specialOptions.style.display = "none";
      } else if (freq === "Special") {
        label.textContent = "Special Payroll Code";
        selectCode.style.display = "none";
        textCode.style.display = "block";
        codeContainer.style.display = "block";
        specialOptions.style.display = "block";
      } else {
        codeContainer.style.display = "none";
        specialOptions.style.display = "none";
      }
    }

    function startPayroll() {
      const freq = document.getElementById("payrollFrequency").value;
      let code = "";
      if (freq === "Semi-Monthly") code = document.getElementById("payrollCodeSelect").value;
      else if (freq === "Special") code = document.getElementById("payrollCodeText").value.trim() || "SPECIAL";
      else if (freq === "Monthly")
    code = "MONTHLY";   // ← FIX


      const entity = document.getElementById("entity").value;
      const selectedGroups = getSelectedGroups();

      if (!entity) return alert("Please select an Entity.");
      if (!selectedGroups.length) return alert("Please select at least one Payroll Group.");
      if (!document.getElementById("startDate").value || !document.getElementById("endDate").value)
        return alert("Please fill out both Start and End dates.");

      const form = {
        payrollFrequency: freq,
        payrollCode: code,
        entity: entity,
        payrollGroups: selectedGroups,
        startDate: document.getElementById("startDate").value,
        endDate: document.getElementById("endDate").value,
        creditingDate: document.getElementById("creditingDate").value,
        computeTax: document.getElementById("computeTax").checked,
        computeContrib: document.getElementById("computeContrib").checked
      };

                 // Save for modal actions
      window.__lastPayrollForm = form;

      // Instead of ugly confirm(), show a clean modal with a table
      google.script.run
        .withSuccessHandler(res => {
          const rows = (res && res.rows) ? res.rows : [];
          if (!rows.length) {
            proceedPayroll(form);
            return;
          }
          openCbpModal(res);
        })
        .withFailureHandler(err => {
          alert("Error checking Computed Basic Pay:\n" + (err.message || err));
        })
        .listMasterfileComputedBasicPayRows(form);
}

function openCbpModal(res) {
  const headerName = res.headerName || "Computed Basic Pay";
  const sheetName  = res.sheetName || "Masterfile";
  const rows = Array.isArray(res.rows) ? res.rows : [];

  document.getElementById('cbpMeta').textContent =
    `Sheet: ${sheetName} · Column: ${headerName} · Rows: ${rows.length}`;

  const tb = document.getElementById('cbpTbody');
  tb.innerHTML = "";

  rows.forEach(r => {
    const rn = Number(r.rowNumber || 0);
    const empId = String(r.empId || "").trim();
    const empName = String(r.empName || "").trim();

    const comp = (typeof r.computed === "number")
      ? r.computed.toLocaleString(undefined, { minimumFractionDigits: 2, maximumFractionDigits: 2 })
      : String(r.computed == null ? "" : r.computed);

    const tr = document.createElement('tr');
    tr.dataset.rowNumber = String(rn);

    tr.innerHTML = `
      <td>${escapeHtml(empId)}</td>
      <td>${escapeHtml(empName)}</td>
      <td style="text-align:right;">${escapeHtml(comp)}</td>
      <td>
        <select class="cbpAction" data-row="${rn}">
          <option value="KEEP" selected>Keep</option>
          <option value="DELETE">Delete</option>
        </select>
      </td>
    `;
    tb.appendChild(tr);
  });

  document.getElementById('cbpModal').style.display = 'flex';
}


function closeCbpModal() {
  document.getElementById('cbpModal').style.display = 'none';
}

function deleteMarkedCbpAndProceed() {
  const form = window.__lastPayrollForm;
  if (!form) return;

  const picked = Array.from(document.querySelectorAll('.cbpAction'))
    .filter(sel => String(sel.value || '') === 'DELETE')
    .map(sel => Number(sel.dataset.row || 0))
    .filter(n => isFinite(n) && n >= 2);

  // If nothing marked DELETE, just proceed
  if (!picked.length) {
    closeCbpModal();
    proceedPayroll(form);
    return;
  }

  google.script.run
    .withSuccessHandler(() => {
      closeCbpModal();
      proceedPayroll(form);
    })
    .withFailureHandler(err => {
      alert("Error deleting Computed Basic Pay:\n" + (err.message || err));
    })
    .clearMasterfileComputedBasicPayRows({ rowNumbers: picked });
}


function cancelPayrollStart() {
  closeCbpModal();
  // stay on form (do nothing)
}


function escapeHtml(s) {
  return String(s || "")
    .replace(/&/g, "&amp;")
    .replace(/</g, "&lt;")
    .replace(/>/g, "&gt;")
    .replace(/"/g, "&quot;")
    .replace(/'/g, "&#039;");
}
function proceedPayroll(form) {
  document.getElementById("formSection").style.display = "none";
  document.getElementById("progressSection").style.display = "block";
  startTime = new Date();
  updateProgress(0, "Starting payroll...", "--");
  progressTimer = setInterval(fetchProgress, 1000);

  google.script.run
    .withSuccessHandler(() => {
      // Backend completion is detected when getProgress() reaches 100%.
    })
    .withFailureHandler(err => {
      updateProgress(100, "❌ Error: " + (err.message || err), "--");
      clearInterval(progressTimer);
    })
    .runPayrollProgressive(form);
}


    function fetchProgress() {
      google.script.run
        .withSuccessHandler(progress => {
          if (!progress) return;
          const { percent, message, eta } = progress;
          updateProgress(percent, message, eta);
          if (percent >= 100) {
  clearInterval(progressTimer);
  setTimeout(() => google.script.host.close(), 2500);
}

        })
        .getProgress();
    }

    function updateProgress(percent, message, eta) {
      const now = new Date();
      const elapsedSec = startTime ? ((now - startTime) / 1000).toFixed(1) : "0.0";
      const bar = document.getElementById("progressBar");
      bar.style.width = percent + "%";
      bar.textContent = percent + "%";
      document.getElementById("status").textContent = message || "";
      document.getElementById("timers").textContent = `Elapsed: ${elapsedSec}s · ETA: ${eta || "--"}`;
    }
