  <script>
    window.onload = function() {
      google.script.run.withSuccessHandler(data => {
        if (!data) return;
        for (const key in data) {
          const el = document.getElementById(key);
          if (el) el.value = data[key];
        }
      }).loadSettings();
    };

    function gatherForm() {
      const form = {};
      document.querySelectorAll("input, select").forEach(el => {
        if (!el.id) return;
        form[el.id] = el.value;
      });
      return form;
    }

    function save() {
      const form   = gatherForm();
      const result = document.getElementById("result");
      const btn    = document.querySelector("button.primary");

      result.textContent = "üíæ Saving settings...";
      if (btn) btn.disabled = true;

      google.script.run
        .withSuccessHandler(ok => {
          if (ok) {
            result.textContent = "‚úÖ Settings saved successfully!";
            // dialog closes; spreadsheet toast is handled on server
            setTimeout(() => google.script.host.close(), 800);
          } else {
            result.textContent = "‚ö†Ô∏è Settings did not save (no confirmation returned).";
            if (btn) btn.disabled = false;
          }
        })
        .withFailureHandler(err => {
          result.textContent = "‚ùå Error saving settings: " + (err && err.message ? err.message : err);
          if (btn) btn.disabled = false;
        })
        .saveSettings(form);
    }

    function resetDefaults() {
      if (!confirm("Are you sure you want to reset all values to defaults?")) return;
      const result = document.getElementById("result");
      result.textContent = "üîÑ Resetting to default values...";

      google.script.run.withSuccessHandler(data => {
        for (const key in data) {
          const el = document.getElementById(key);
          if (el) el.value = data[key];
        }
        result.textContent = "üîÑ Reset to default values.";
      }).resetSettingsToDefaults();
    }

    function testConnections() {
      const result = document.getElementById('result');
      result.innerText = 'üï∞Ô∏è Testing connections...';
      const form = gatherForm();

      google.script.run
        .withSuccessHandler((results) => {
          const msg = `
üîç Connection Test Results:
Masterfile: ${results.masterfile}
Adjustments: ${results.adjustments}
          `;
          result.textContent = msg.trim();
        })
        .withFailureHandler((err) => {
          result.textContent = `‚ùå Error testing connections: ${err.message || err}`;
        })
        .testConnections(form);
    }
  </script>
